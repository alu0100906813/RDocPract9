def at(i)
    if(@head_==nil || i<0)
        nil
    end
    auxn = @head_
    iter = 0
    while auxn!=nil do
        if iter==i
            return auxn
        end
        auxn = auxn["next"]
        iter=iter+1
    end
    nil
end

def get_gei()
    if @head_ == nil
        nil # Lista está vacía
    else
        total_gei = 0.0
        auxn = @head_
        while auxn!=nil do
            total_gei = total_gei+auxn["value"].get_gei()
            auxn = auxn["next"]
        end
        total_gei.truncate(1) # Devolvemos el total del GEI
    end
end

def push(value)
    naux = Nodo.new(value, nil, nil)
    if @head_ == nil
        @head_ = naux
        @tail_ = naux
    else
        aux = @tail_
        while(aux["next"]!=nil) do
            aux = aux["next"]
        end
        naux["prev"] = aux
        aux["next"] = naux
        @tail_ = naux
    end
end

def size()
    if @head_ == nil
        0 # Está vacía
    else
        aux = 1
        naux = @head_
        while naux["next"]!=nil do
            aux = aux+1
            naux = naux["next"]
        end
        aux # Retornamos el tamaño
    end
end
