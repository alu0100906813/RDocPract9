    # File lib/eficiencia.rb, line 25
def emisiones_gei()
    if @alimentos.get_head()==nil
        return 0.0
    else
        gei = 0.0
        auxn = @alimentos.get_head()
        i = 0
        while auxn!=nil do
            gei += auxn["value"].get_gei()*(cantidad[i]/100)
            auxn = auxn["next"]
            i += 1
        end
        gei.truncate(2)
    end
end
